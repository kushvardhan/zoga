    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // Get the demo container
            const demo = document.getElementById('web-design-demo');
            
            // Get the elements by their NEWLY ADDED IDs to avoid selector issues
            const primaryBar = document.getElementById('primary-bar');
            const buttonBar = document.getElementById('button-bar');
            
            // Safety check: ensure both elements exist before proceeding
            if (!primaryBar || !buttonBar) {
                console.error("Critical elements for Web Design demo not found. Check IDs 'primary-bar' and 'button-bar'.");
                return; 
            }

            const paletteDots = document.querySelectorAll('.palette-dot');

            // Function to handle palette click
            function changePalette(event) {
                const dot = event.target;
                const primary = dot.dataset.primary;
                const secondary = dot.dataset.secondary;

                // Update demo colors
                primaryBar.style.backgroundColor = primary;
                buttonBar.style.backgroundColor = secondary;
            }

            // Attach click listener to all palette dots
            paletteDots.forEach(dot => {
                dot.addEventListener('click', changePalette);
            });
        });
    </script>
